# Практика по созданию статических методов

```js
class Comment {
  constructor(text) {
    this.text = text;
    this.votesQty = 0;
  }

  upvote() {
    this.votesQty += 1;
  }

  static mergeComments(first, second) {
    return `${first} ${second}`;
  }
}

Comment;

console.dir(Comment);

Comment.prototype;

const myComment = new Comment("Awesome comment");

myComment;

Comment.mergeComments("Some text.", "Some other text");
```
