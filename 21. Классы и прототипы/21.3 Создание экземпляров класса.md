# Создание экземпляров класса

```js
class Comment {
  constructor(text) {
    // Переменная this указывает на экземпляр класса
    this.text = text;
    this.votesQty = 0;
  }

  upvote() {
    this.votesQty += 1;
  }
}

// Вызывается функция constructor
const firstComment = new Comment("First comment");

// Собственные свойства экземпляра, унаследованные свойства экземпляра, наследование по цепочке
console.log(firstComment);
```
