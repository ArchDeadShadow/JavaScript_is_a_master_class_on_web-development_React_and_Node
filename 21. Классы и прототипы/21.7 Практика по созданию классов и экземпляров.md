# Практика по созданию классов и экземпляров

```js
class Comment {
  constructor(text) {
    this.text = text;
    this.votesQty = 0;
  }

  upvote() {
    this.votesQty += 1;
  }
}

// Comment.prototype

const firstComment = new Comment("First comment");

const myObject = {};

// myObject.constructor

firstComment.upvote();

firstComment.votesQty;
```
